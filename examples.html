Joystick Control

Control your XRP with a gamepad using arcade-style driving. Connect via Bluetooth, then use the left joystick to drive forward/backward and turn.

joystick_control.py 
from XRPLib.defaults import *
from XRPLib.gamepad import *
import time

# Get the default gamepad instance
gp = Gamepad.get_default_gamepad()
print("Starting!")

while True:
    # Arcade drive: one joystick controls everything
    throttle = gp.get_value(gp.Y1)  # Forward/backward on left stick
    rotation = gp.get_value(gp.X1)  # Turn left/right on left stick

    # Use arcade method for intuitive control
    drivetrain.arcade(throttle, rotation)
    time.sleep(0.2)
